<script>
  // Tooltip toggling for touch devices
  function setupTooltips(){
    const infos=[...document.querySelectorAll('.info')];
    function closeAll(){infos.forEach(i=>i.classList.remove('active'));};
    infos.forEach(i=>{
      i.addEventListener('click',e=>{
        e.stopPropagation();
        const active=i.classList.toggle('active');
        if(active){infos.filter(x=>x!==i).forEach(x=>x.classList.remove('active'));}});
      i.addEventListener('keydown',e=>{
        if(e.key==='Enter'||e.key===' '){e.preventDefault();i.click();}});
    });
    document.addEventListener('click',closeAll);
  }
  setupTooltips();
</script>
